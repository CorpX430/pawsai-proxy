function sr(str, n) {
let h = n * 2654435761;
for (let i = 0; i < str.length; i++) h = Math.imul(h ^ str.charCodeAt(i), 2654435761);
return ((h >>> 0) % 1000) / 1000;
}
function predict(home, away) {
const k = `${home}|${away}`;
const hw = Math.round(28 + sr(k,1)*38);
const dr = Math.round(14 + sr(k,2)*22);
const aw = Math.max(5, 100 - hw - dr);
const hg = (0.7 + sr(k,3)*1.8).toFixed(1);
const ag = (0.5 + sr(k,4)*1.5).toFixed(1);
const btts= Math.round(38 + sr(k,5)*35);
const o25 = Math.round(42 + sr(k,6)*32);
const conf= Math.round(52 + sr(k,7)*38);
const tips= [
{ label:“Home Win”, value:hw, odds:(100/hw*0.88).toFixed(2) },
{ label:“Draw”,     value:dr, odds:(100/dr*0.88).toFixed(2) },
{ label:“Away Win”, value:aw, odds:(100/aw*0.88).toFixed(2) },
];
const best = tips.reduce((a,b) => a.value > b.value ? a : b);
return { hw, dr, aw, hg, ag, btts, o25, conf, best };
}

function uiStatus(m) {
const s = m.status;
if ([“IN_PLAY”,“PAUSED”].includes(s)) return “LIVE”;
if (s === “HALFTIME”) return “HT”;
if (s === “FINISHED”) return “FT”;
return “UPCOMING”;
}
function liveMinute(m) { return m.minute ?? m.score?.minute ?? null; }
function getScore(m) {
const s = m.score;
if (!s) return null;
if ([“IN_PLAY”,“PAUSED”,“HALFTIME”].includes(m.status)) {
const r = s.regular ?? s.fullTime;
return r?.home != null ? r : null;
}
if (m.status === “FINISHED”) return s.fullTime?.home != null ? s.fullTime : null;
return null;
}
function kickoffTime(m) {
if (!m.utcDate) return “–:–”;
return new Date(m.utcDate).toLocaleTimeString(“en-GB”,{hour:“2-digit”,minute:“2-digit”});
}
function kickoffDate(m) {
if (!m.utcDate) return “”;
const d = new Date(m.utcDate), today = new Date();
const diff = Math.round((d - today) / 86400000);
if (diff === 0) return “Today”;
if (diff === 1) return “Tomorrow”;
if (diff === -1) return “Yesterday”;
return d.toLocaleDateString(“en-GB”,{weekday:“short”,day:“numeric”,month:“short”});
}
